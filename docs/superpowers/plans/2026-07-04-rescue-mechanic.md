# Rescue Mechanic Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the boat rescue NPC mechanic: when the boat touches an NPC, the NPC is consumed and the boat gains one crew member (up to a max of 5), with the HUD showing current crew count.

**Architecture:** Add an `Area2D` rescue trigger to `NPC.tscn` that calls a new `Boat.gain_crew()` method. Extend `Boat` to clamp `crew_count` at `max_crew_count`. Extend the HUD to display `Crew: current/max` by listening to `Boat.crew_count_changed`.

**Tech Stack:** Godot 4.7, GDScript, Jolt Physics 2D.

## Task 2: Create NPC rescue trigger script

**Files:**
- Create: `scripts/characters/npc_rescue.gd`
- Create: `scripts/characters/npc_rescue.gd.uid` (generated by Godot)

**Interfaces:**
- Consumes: `body.gain_crew(amount: int)` on objects in group `"boats"`.
- Produces: nothing exported to other scripts.

- [ ] **Step 1: Create `scripts/characters/npc_rescue.gd`**

```gdscript
class_name NPCRescue
extends Area2D

@export var rescue_value: int = 1

var _rescued: bool = false

func _ready() -> void:
	body_entered.connect(_on_body_entered)


func _on_body_entered(body: Node2D) -> void:
	if _rescued:
		return

	if not body.is_in_group("boats"):
		return

	if not is_instance_valid(body):
		return

	if not body.has_method("gain_crew"):
		return

	_rescued = true
	body.gain_crew(rescue_value)
	queue_free()
```

## Task 4: Display crew count on HUD

**Files:**
- Modify: `scenes/ui/HUD.tscn`
- Modify: `scripts/ui/hud.gd`

**Interfaces:**
- Consumes: `Boat.crew_count_changed(count: int)` signal and `Boat.max_crew_count` property.
- Produces: `CrewLabel` text updates.

- [ ] **Step 1: Add `CrewLabel` to `scenes/ui/HUD.tscn`**

Open `scenes/ui/HUD.tscn` in Godot. Under `MarginContainer/VBoxContainer`, add a new `Label` named `CrewLabel` after `StatusLabel`. Set its text to `Crew: 3/5`.

Make it a unique name (`CrewLabel`) so it can be referenced with `%CrewLabel`.

- [ ] **Step 2: Modify `scripts/ui/hud.gd`**

Replace the contents with:

```gdscript
extends CanvasLayer

@onready var score_label: Label = %ScoreLabel
@onready var status_label: Label = %StatusLabel
@onready var crew_label: Label = %CrewLabel


func _ready() -> void:
	GameState.score_changed.connect(_on_score_changed)
	GameState.pause_changed.connect(_on_pause_changed)
	EventBus.player_spawned.connect(_on_player_spawned)

	_on_score_changed(GameState.score)
	_on_pause_changed(GameState.is_paused)

	var boat := _find_boat()
	if boat != null:
		_update_crew_label(boat.crew_count, boat.max_crew_count)
		boat.crew_count_changed.connect(_on_crew_count_changed)


func _find_boat() -> Boat:
	var boats := get_tree().get_nodes_in_group("boats")
	for node in boats:
		if node is Boat:
			return node
	return null


func _update_crew_label(count: int, max_count: int) -> void:
	crew_label.text = "Crew: %d/%d" % [count, max_count]


func _on_score_changed(score: int) -> void:
	score_label.text = "Score: %s" % score


func _on_pause_changed(is_paused: bool) -> void:
	status_label.text = "Paused" if is_paused else "Running"


func _on_crew_count_changed(count: int) -> void:
	var boat := _find_boat()
	if boat == null:
		return
	_update_crew_label(count, boat.max_crew_count)


func _on_player_spawned(_player: Node) -> void:
	var boat := _find_boat()
	if boat == null:
		return
	if not boat.crew_count_changed.is_connected(_on_crew_count_changed):
		boat.crew_count_changed.connect(_on_crew_count_changed)
	_update_crew_label(boat.crew_count, boat.max_crew_count)
```
